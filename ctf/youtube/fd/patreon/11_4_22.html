from PIL import Image, ImageColor

with open('code.txt') as my_file:
    linearray = [line.strip() for line in my_file]

to_arr = [[char for char in llin]for llin in linearray]

img = Image.new(mode = "RGB", size = (to_arr.__len__(),to_arr.__len__()))
row=0
line=0

pixel_array=""

for kl in to_arr:
    for ki in kl:
      if ki == '1':
        pixel_array += str("1")
        img.putpixel((row,line), ImageColor.getcolor('black', '1'))
      elif ki == '0':
        pixel_array += str("0")
        img.putpixel((row,line), ImageColor.getcolor('white', '1'))
      else:
        print(ki)
      row=row+1
    line=line+1
    row=0
img.save("qrcode .png")
